#include<stdio.h>
#include<stdlib.h>
#define EMPTY 0

typedef struct node {
	int data;
	struct node*link;
}List;

List*get_node() {
	List*tmp;
	tmp = (List*)malloc(sizeof(List));
	tmp->link = EMPTY;
	return tmp;
}
void INsertNode(List**head, int data) {
	List*tmp;
	tmp = *head;
	if (head == EMPTY) {
		*head = get_node();
		(*head)->data = data;
		(*head)->link = tmp;
	}
	while (tmp->link) {
		tmp = tmp->link;
	}
	tmp->link = get_node();
	tmp->link->data = data;
	tmp->link->link = EMPTY;
}
int DeleteNode(List**head) {
	List*tmp;
	tmp = *head;
	int num;
	if (tmp == EMPTY) {
		return;
	}
	num = tmp->data;
	*head = (*head)->link;
	return num;
	free(tmp);
}

int main()
{
	List*head = EMPTY;
	InsertNode(&head, 10);
	InsertNode(&head, 20);
	InsertNode(&head, 30);

	printf("%d\n", DeleteNode(&head));
	printf("%d\n", DeleteNode(&head));
	printf("%d\n", DeleteNode(&head));
}
