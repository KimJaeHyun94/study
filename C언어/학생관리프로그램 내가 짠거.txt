#include<iostream>
#include<iomanip>
#include<windows.h>
#include<string.h>
using namespace std;

class computer {
	int kor, mat, eng,sum;
	double pung;
	int i = 0;
	computer*cc;
	char name[20];
	int ss;
public:
	computer() {}
	computer(int people){
		cc = new computer[people];
	}
	void Choco(char*n) {
		if (i >= 10) {
			cout << "학생 초과입니다!" << endl;
		}
		else {
			if (i == 0) {
				strcpy(cc[0].name, n);
				i++;
			}
			else {
				bool exist = FALSE;
				for (int k = 0; k < i; k++) {
					if (strcmp(cc[k].name, n) == 0)
					{
						exist = TRUE;
						cout << "같은 이름이 있습니다!" << endl;
						break;
					}
				}
				if (!exist) {
					strcpy(cc[i].name, n);
					i++;
				}
			}
		}
	}
	void Score(char*n) {
		if (i == 0) {
			cout << "등록된 사람이 없습니다!" << endl;
		}
		else {
			bool Exist = FALSE;
			for (int k = 0; k <= i; k++) {
				if (strcmp(cc[k].name, n) == 0) {
					cout << "성적을 입력하세요 :";
					cin >> cc[k].eng >> cc[k].kor >> cc[k].mat;
					cc[k].ss = 1;
					Exist = TRUE;
					break;
				}
			}
			if (!Exist) {
				cout << "맞는 사람이 없습니다!" << endl;
			}
		}
	}
	void Showdata() {
		if (i == 0) {
			cout << "등록된 학생이 없습니다" << endl;
		}
		else {
			system("cls");
			for (int k = 0; k < i; k++) {
				cout << k + 1 << "번째 학생 : " << cc[k].name << endl;
			}
		}
	}
	void Sum(char*n) {
		
		if (i == 0) {
			cout << "등록된 사람이 없습니다!" << endl;
		}
		else {
			bool Exist = FALSE;
			for (int k = 0; k <= i; k++) {
				cc[k].sum = cc[k].kor + cc[k].mat + cc[k].eng;
				if (strcmp(cc[k].name, n) == 0) {
					if (cc[k].ss == 1) {
						cout << "합계는 : " << cc[k].sum << endl;
						Exist = TRUE;
						break;
					}
					else {
						cout << "성적을 입력하세요" << endl;
						Exist = TRUE;
					}
				}
			}
			if (!Exist) {
				cout << "맞는 사람이 없습니다!" << endl;
			}
		}
	}
	void Pung(char*n) {
		if (i == 0) {
			cout << "등록된 사람이 없습니다!" << endl;
		}
		else {
			bool Exist = FALSE;
			for (int k = 0; k <= i; k++) {
				if (strcmp(cc[k].name, n) == 0) {
					cc[k].pung = (cc[k].kor + cc[k].mat + cc[k].eng) / (3.0);
					if (cc[k].ss == 1) {
						cout << "평균은 : " << cc[k].pung << endl;
						Exist = TRUE;
						break;
					}
					else {
						cout << "성적을 입력하세요" << endl;
						Exist = TRUE;
					}
				}
			}
			if (!Exist) {
				cout << "맞는 사람이 없습니다!" << endl;
			}
		}
	}
	~computer() {
		delete[]cc;
	}
};
